import unittest

from Tests import test_Leave_Types
from Tests import test_Leave_Config
from Tests import test_Leave_Plan

def suite():
    test_suite = unittest.TestSuite()

    # Add test cases in the desired order
    test_suite.addTest(unittest.makeSuite(test_Leave_Types))
    test_suite.addTest(unittest.makeSuite(test_Leave_Config))
    test_suite.addTest(unittest.makeSuite(test_Leave_Plan))

    return test_suite


if __name__ == "__main__":

    runner = unittest.TextTestRunner()
    runner.run(suite())
